### Код

```vibee
// Глава 441: AVL-дерево
fn глава_441() {
    let тройка = 3;
    let тридевятое = тройка * тройка * тройка;  // 27
    let священное = 441 % тридевятое;
    return священное == 0;  // благословенное (÷9)
}
```
