## КОД СИСТЕМЫ

```vibee
/// Система достижений
struct Игрок {
    имя: String,
    xp: u32,
    уровень: u8,
    достижения: Vec<Достижение>,
    гильдия: Option<Гильдия>,
    серия_дней: u8,
}

impl Игрок {
    fn добавить_xp(&mut self, количество: u32) {
        // Бонус за серию
        let множитель = match self.серия_дней {
            0..=2 => 1,
            3..=8 => 2,
            9..=26 => 3,
            _ => 9,
        };
        
        self.xp += количество * множитель;
        self.проверить_уровень();
    }
    
    fn проверить_уровень(&mut self) {
        // Уровень = log₃(XP / 3) + 1
        self.уровень = ((self.xp as f64 / 3.0).log(3.0) + 1.0) as u8;
    }
    
    fn титул(&self) -> &str {
        match self.уровень {
            1..=2 => "Странник",
            3..=8 => "Искатель",
            9..=11 => "Ученик",
            12..=17 => "Подмастерье",
            18..=20 => "Мастер",
            21..=26 => "Эксперт",
            27..=32 => "Мудрец",
            33..=80 => "Архимаг",
            81..=242 => "Легенда",
            243..=728 => "Бессмертный",
            729..=998 => "Создатель",
            _ => "Хранитель Тридевятого",
        }
    }
}

/// Проверка достижения
fn проверить_достижение(игрок: &Игрок, действие: Действие) -> Option<Достижение> {
    match действие {
        Действие::ПрочитатьГлаву(1) => Some(Достижение::ПервыйШаг),
        Действие::ПрочитатьГлаву(n) if n % 9 == 0 => Some(Достижение::Книгочей),
        Действие::НаписатьКод if игрок.код.contains('Ⲅ') => Some(Достижение::ДругТройки),
        // ...
        _ => None,
    }
}
```
