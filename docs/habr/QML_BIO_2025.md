## Часть VII: Код для Проверки

### Python

```python
import math

PHI = (1 + math.sqrt(5)) / 2

def verify_golden_codon():
    """φ² + 1/φ² = 3 = codon length"""
    phi_sq = PHI ** 2
    phi_inv_sq = 1 / phi_sq
    result = phi_sq + phi_inv_sq
    codon_length = 3
    print(f"φ² + 1/φ² = {result}")
    print(f"Codon length = {codon_length}")
    print(f"Match: {abs(result - codon_length) < 1e-10}")
    return abs(result - codon_length) < 1e-10

def lucas(n):
    """L(n) = φⁿ + 1/φⁿ"""
    return PHI**n + (1/PHI)**n

def encode_dna(sequence):
    """Encode DNA to angles"""
    bases = {'A': 0, 'T': 1, 'G': 2, 'C': 3}
    return [bases[b] * math.pi / 2 for b in sequence]

def translate_codon(codon):
    """Simplified codon translation"""
    bases = {'A': 0, 'T': 1, 'G': 2, 'C': 3}
    idx = bases[codon[0]] * 16 + bases[codon[1]] * 4 + bases[codon[2]]
    return idx % 20

# Проверки
print("=== GOLDEN CODON ===")
verify_golden_codon()

print("\n=== LUCAS NUMBERS ===")
print(f"L(3) = {lucas(3):.0f} = DNA bases")

print("\n=== DNA ENCODING ===")
seq = "ATG"  # Start codon
angles = encode_dna(seq)
print(f"ATG → {angles}")

print("\n=== CODON TRANSLATION ===")
print(f"ATG → amino acid {translate_codon('ATG')}")
```
