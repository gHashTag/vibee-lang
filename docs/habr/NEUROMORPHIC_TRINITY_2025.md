import math

PHI = (1 + math.sqrt(5)) / 2

def lif_neuron(membrane, input_current, dt, tau=PHI):
    """Leaky Integrate-and-Fire с τ = φ"""
    decay = math.exp(-dt / tau)
    new_membrane = membrane * decay + input_current
    return new_membrane

def encode_ternary(value):
    """Тернарное кодирование (0, 1, 2)"""
    if value < 0.33:
        return 0
    elif value < 0.67:
        return 1
    else:
        return 2

def verify_golden_trinity():
    """φ² + 1/φ² = 3 = spike levels"""
    phi_sq = PHI ** 2
    phi_inv_sq = 1 / phi_sq
    result = phi_sq + phi_inv_sq
    print(f"φ² + 1/φ² = {result}")
    print(f"Spike levels = 3")
    print(f"Match: {abs(result - 3) < 1e-10}")
    return abs(result - 3) < 1e-10

def decompose_603():
    """603 = 67 × 3²"""
    n = 603
    k = 0
    while n % 3 == 0:
        n //= 3
        k += 1
    print(f"603 = {n} × 3^{k}")
    return n, k

# Проверки
print("=== GOLDEN TRINITY ===")
verify_golden_trinity()

print("\n=== LIF NEURON ===")
membrane = 0.0
for i in range(10):
    membrane = lif_neuron(membrane, 0.5, 0.1)
    print(f"Step {i}: membrane = {membrane:.4f}")

print("\n=== TERNARY ENCODING ===")
for v in [0.1, 0.5, 0.9]:
    print(f"{v} -> level {encode_ternary(v)}")

print("\n=== 603 DECOMPOSITION ===")
decompose_603()
