actor.vibee 
/// Actor-based counter
actor Counter {
    count: Int,
    
    pub fn increment(mut self) {
        self.count += 1
    }
    
    pub fn get(self) -> Int {
        self.count
    }
}

async fn main() {
    let counter = spawn Counter { count: 0 }
    
    counter.increment().await
    counter.increment().await
    
    let value = counter.get().await
    println("Count: {}", value)
}
